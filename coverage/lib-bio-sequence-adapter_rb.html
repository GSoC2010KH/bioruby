33   # Defines a reader attribute method with psudo lazy evaluation/memoization.
 
34   #
 
35   # It defines a method <i>name</i> like attr_reader, but at the first time
 
36   # when the method <i>name</i> is called, it acts as follows:
 
37   # When instance variable @<i>name</i> is not defined,
 
38   # calls <tt>__get__<i>name</i>(@source_data)</tt> and stores the returned
 
39   # value to @<i>name</i>, and changes its behavior to the same as
 
40   # <tt>attr_reader </tt><i>:name</i>.
 
41   # When instance variable @name is already defined,
 
42   # its behavior is changed to the same as
 
43   # <tt>attr_reader </tt><i>:name</i>.
 
44   # When the object is frozen, storing to the instance variable and
 
45   # changing methods behavior do not occur, and the value of
 
46   # <tt>__get__<i>name</i>(@source_data)</tt> is returned.
 
47   # 
48   # Note that it assumes that the source data object is stored in
 
49   # @source_data instance variable.
 
50   def attr_reader_lazy(name)
 
51     #$stderr.puts "attr_reader_lazy :#{name}"
 
52     varname = "@#{name}".intern
 
53     methodname = "__get__#{name}".intern
 
54 
55     # module to reset method's behavior to normal attr_reader
 
56     reset = "Attr_#{name}".intern
 
57     const_set(reset, Module.new { attr_reader name })
 
58     reset_module_name = "#{self}::#{reset}"
 
59 
60     # define attr method
 
61     module_eval <<__END_OF_DEF__
 
62       def #{name}
 
63         unless defined? #{varname} then
 
64           #$stderr.puts "LAZY #{name}: calling #{methodname}"
 
65           val = #{methodname}(@source_data)
 
66           #{varname} = val unless frozen?
 
67         else
 
68           val = #{varname}
 
69         end
 
70         unless frozen? then
 
71           #$stderr.puts "LAZY #{name}: finalize: attr_reader :#{name}"
 
72           self.extend(#{reset_module_name})
 
73         end
 
74         val
 
75       end
 
76 __END_OF_DEF__
 
77   end

79   # Defines a Bio::Sequence to Bio::* adapter method with
 
80   # psudo lazy evaluation and psudo memoization.
 
81   #
 
82   # Without block, defines a private method <tt>__get__<i>name</i>(orig)</tt>
 
83   # which calls <i>source_method</i> for @source_data.
 
84   #
 
85   # def__get__(name, source_method) is the same as:
 
86   #   def __get__name(orig); orig.source_method; end
 
87   #   attr_reader_lazy name
 
88   #
 
89   # If block is given, <tt>__get__<i>name</i>(orig)</tt> is defined
 
90   # with the block. The @source_data is given as an argument of the block,
 
91   # i.e. the block must get an argument.
 
92   #
 
93   def def_biosequence_adapter(name, source_method = name, &block)
 
94     methodname = "__get__#{name}".intern
 
95 
96     if block then
 
97       define_method(methodname, block)
 
98     else
 
99       module_eval <<__END_OF_DEF__
 
100         def #{methodname}(orig)
