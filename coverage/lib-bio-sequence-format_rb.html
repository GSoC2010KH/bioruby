143   # Using Bio::Sequence::Format, return a String with the Bio::Sequence
 
144   # object formatted in the given style.
 
145   #
 
146   # Formats currently implemented are: 'fasta', 'genbank', and 'embl'
 
147   #
 
148   #   s = Bio::Sequence.new('atgc')
 
149   #   puts s.output(:fasta)                   #=> "> \natgc\n"
 
150   #
 
151   # The style argument is given as a Ruby 
152   # Symbol(http://www.ruby-doc.org/core/classes/Symbol.html)
 
153   # ---
 
154   # *Arguments*: 
155   # * (required) _format_: :fasta, :genbank, *or* :embl
 
156   # *Returns*:: String object
 
157   def output(format = :fasta, options = {})
 
158     formatter_const = format.to_s.capitalize.intern
 
159 
160     formatter_class = nil
 
161     get_formatter_repositories.each do |mod|
 
162       begin
 
163         formatter_class = mod.const_get(formatter_const)
 
164       rescue NameError
 
165       end
 
166       break if formatter_class
 
167     end
 
168     unless formatter_class then
 
169       raise "unknown format name #{format.inspect}"
 
170     end
 
171 
172     formatter_class.output(self, options)
 
173   end

269   # format qualifiers
 
270   def format_qualifiers(qualifiers, indent, width)
 
271     qualifiers.collect do |qualifier|
 
272       q = qualifier.qualifier
 
273       v = qualifier.value.to_s
 
274 
275       if v == true
 
276         lines = wrap_with_newline('/' + q, width)
 
277       elsif q == 'translation'
 
278         lines = fold("/#{q}=\"#{v}\"", width)
 
279       else
 
280         if v[/\D/] or q == 'chromosome'
 
281           #v.delete!("\x00-\x1f\x7f-\xff")
 
282           v.gsub!(/"/, '""')
 
283           v = '"' + v + '"'
 
284         end
 
285         lines = wrap_with_newline('/' + q + '=' + v, width)
 
286       end
 
287 
288       lines.gsub!(/^/, indent)
 
289       lines
 
290     end.join
 
291   end

301   def wrap_and_split_lines(str, width)
 
302     result = []
 
303     lefts = str.chomp.split(/(?:\r\n|\r|\n)/)
 
304     lefts.each do |left|
 
305       left.rstrip!
 
306       while left and left.length > width
 
307         line = nil
 
308         width.downto(1) do |i|
 
309           if left[i..i] == ' ' or /[\,\;]/ =~ left[(i-1)..(i-1)]  then
 
310             line = left[0..(i-1)].sub(/ +\z/, '')
 
311             left = left[i..-1].sub(/\A +/, '')
 
312             break
 
313           end
 
314         end
 
315         if line.nil? then
 
316           line = left[0..(width-1)]
 
317           left = left[width..-1]
 
318         end
 
319         result << line
 
320         left = nil if  left.to_s.empty?
 
321       end
 
322       result << left if left
 
323     end
 
324     return result
 
325   end

327   def wrap_with_newline(str, width)
 
328     result = wrap_and_split_lines(str, width)
 
329     result_string = result.join("\n")
 
330     result_string << "\n" unless result_string.empty?
 
331     return result_string
 
332   end

334   def wrap(str, width = 80, prefix = '')
 
335     actual_width = width - prefix.length
 
336     result = wrap_and_split_lines(str, actual_width)
 
337     result_string = result.join("\n#{prefix}")
 
338     result_string = prefix + result_string unless result_string.empty?
 
339     return result_string
 
340   end
