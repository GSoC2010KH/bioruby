80     def fastq_quality_string(seq, default_score)
 
81       sc = fastq_quality_scores(seq)
 
82       if sc.size < seq.length then
 
83         if default_score == true then
 
84           # when true, the highest score
 
85           default_score = fastq_format_data.score_range.end
 
86         else
 
87           # when false or nil, the lowest score
 
88           default_score ||= fastq_format_data.score_range.begin
 
89         end
 
90         sc = sc + ([ default_score ] * (seq.length - sc.size))
 
91       end
 
92       fastq_format_data.scores2str(sc)
 
93     end
