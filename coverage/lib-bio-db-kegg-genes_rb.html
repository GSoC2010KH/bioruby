308   # *Returns*:: Hash
 
309   def codon_usage(codon = nil)
 
310     unless @data['CODON_USAGE']
 
311       hash = Hash.new
 
312       list = cu_list
 
313       base = %w(t c a g)
 
314       base.each_with_index do |x, i|
 
315         base.each_with_index do |y, j|
 
316           base.each_with_index do |z, k|
