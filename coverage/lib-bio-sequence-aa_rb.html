47   # Amino Acid sequences are *always* all uppercase in bioruby
 
48   #
 
49   #   s = Bio::Sequence::AA.new("rrLeHtfV")
 
50   #   puts s                                  #=> "RRLEHTFVF"
 
51   #
 
52   # Whitespace is stripped from the sequence
