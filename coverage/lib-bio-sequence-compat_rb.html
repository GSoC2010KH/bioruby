40   # <b>Bio::Sequence#to_fasta is DEPRECIATED</b>
 
41   # Do not use Bio::Sequence#to_fasta ! Use Bio::Sequence#output instead. 
42   # Note that Bio::Sequence::NA#to_fasta, Bio::Sequence::AA#to_fasata,
 
43   # and Bio::Sequence::Generic#to_fasta <b>can still be used</b>,
 
44   # because there are no alternative methods.
 
45   # 
46   # Output the FASTA format string of the sequence.  The 1st argument is
 
47   # used as the comment string.  If the 2nd option is given, the output
 
48   # sequence will be folded.
 
49   # ---
 
50   # *Arguments*:
 
51   # * (optional) _header_: String object
 
52   # * (optional) _width_: Fixnum object (default nil)
 
53   # *Returns*:: String
 
54   def to_fasta(header = '', width = nil)
 
55     warn "Bio::Sequence#to_fasta is obsolete. Use Bio::Sequence#output(:fasta) instead" if $DEBUG
 
56     ">#{header}\n" +
 
57     if width
 
58       self.to_s.gsub(Regexp.new(".{1,#{width}}"), "\\0\n")
 
59     else
 
60       self.to_s + "\n"
 
61     end
 
62   end
